CREATE TABLE Bilancio (codice_siope TEXT NOT NULL, codice_ente TEXT REFERENCES Ente (codice_ente) ON DELETE CASCADE ON UPDATE CASCADE MATCH SIMPLE NOT NULL, anno INTEGER NOT NULL, descrizione_codice TEXT, importo REAL NOT NULL, PRIMARY KEY ( codice_siope ASC, codice_ente ASC, anno ASC ));
CREATE TABLE Appalti (cig TEXT NOT NULL, oggetto TEXT NOT NULL, scelta_contraente TEXT, codice_fiscale_aggiudicatario TEXT NOT NULL, aggiudicatario TEXT NOT NULL, importo REAL NOT NULL, codice_ente TEXT REFERENCES Ente (codice_ente) ON DELETE CASCADE ON UPDATE CASCADE MATCH SIMPLE NOT NULL, PRIMARY KEY ( cig ASC, oggetto ASC, codice_fiscale_aggiudicatario ASC ));
